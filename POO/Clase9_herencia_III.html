class Punto2():
    
    # Constructor
    def __init__(self, x, y):
        self.x = x
        self.y = y
        
    # __str__
    def __str__(self):
        return f'{self.x, self.y}'
    
    # Mtodo norma:
    def auxiliar(self):
        return self.x ** 2 + self.y ** 2
    
    def norma2(self):
        return np.sqrt(self.auxiliar())

y pasemos a modificar el cdigo de la subclase Punto3 de modo que las clases Punto2 y Graficador3 sean sus respectivas superclases. Para ello

  1. Al momento de definir la subclase Punto3 pondremos como parmetros los nombres de las superclases. Es decir, en vez de escribir class Punto3(Punto2) como lo habamos escrito en sesiones anteriores pondremos class Punto3(Punto2, Graficador3)

  2. En herencia mltiple no utilizaremos el super(). para referenciar mtodos de las superclases, pues en estos casos tendremos ms de una superclase en cuestin. En su lugar usaremos nombre_de_la_superclase().

# Subclase Punto3 de las superclase Punto2 y Graficador3

class Punto3(Punto2, Graficador3):
    
    # Constructor
    def __init__(self, x, y, z):
        
        # Hacemos referencia de los parmetros del mtodo __init__ de la superclase
        # Punto2:
        
        Punto2.__init__(self, x, y)
        
        # Atributo propio de la subclase Punto3
        self.z = z
        
    # __str__
    def __str__(self):
        
        # Hacemos referencia de la salida del mtodo __str__ de la superclase 
        # Punto2:
        
        return f'({Punto2.__str__(self)},{self.z})'
    
    # Mtodo norma
    def auxiliar(self):
        return Punto2.auxiliar() + self.z ** 2
    
    def norma3(self):
        return np.sqrt(self.auxiliar())

# Subclase Cuadrado de las superclases Figura y Color

class Cuadrado(Figura, Color):
    
    # Constructor
    def __init__(self, lado, color):
        
        # Sobreescritura del mtodo __init__ correspondiente a la clase Figura
        Figura.__init__(self, lado, lado)
        
        # Sobreescritura del mtodo __init__ correspondiente a la clase Color
        Color.__init__(self, color)
        
    # Mtodo para calcular el rea
    def area(self):
        return self.ancho * self.altura

donde podemos observar que escribimos self.ancho * self.altura en vez de intentar escribir self.lado * self.lado. La razn radica en que tanto self.ancho como self.altura son atributos heredados de la clase Figura y el nombre de lado slo hace alusin al nombre del parmetro que pasamos en el constructor de la clase Cuadrado.
